292template < class T, class L, class A, size_t ... Ds > requires ( ( Ds >= 0 ) && ... )
293template < class Lambda >
294constexpr fs_tensor<T,L,A,Ds...>::fs_tensor( Lambda&& lambda ) noexcept( noexcept( declval<Lambda&&>()( Ds ... ) ) )
295 requires requires { { declval<Lambda&&>().operator()( Ds ... ) } -> convertible_to<typename fs_tensor<T,L,A,Ds...>::element_type>; }
296{
297 // If expression is no except, then no need to capture last exception
298 static const bool lambda_is_noexcept = is_nothrow_convertible_v< decltype( declval<Lambda&&>()( Ds ... ) ), element_type >;
299 if constexpr ( lambda_is_noexcept )
300 {
301 // Construct all elements from lambda output
302 auto ctor = [this,&lambda]< class ... SizeType >( SizeType ... indices ) constexpr noexcept
303 {
304 this->underlying_span()[ indices ... ] = lambda( indices ... );
305 };
306 detail::apply_all( this->underlying_span(), ctor, execution::unseq );
307 }
308 else
309 {
310 // Cache the last exception to be thrown
311 exception_ptr eptr;
312 // Construct all elements from lambda output
313 auto ctor = [this,&lambda,&eptr]< class ... SizeType >( SizeType ... indices ) constexpr noexcept
314 {
315 try
316 {
317 this->underlying_span()(indices...) = lambda(indices...);
318 }
319 catch ( ... )
320 {
321 eptr = current_exception();
322 }
323 };
324 // Apply lambda
325 detail::apply_all( this->underlying_span(), ctor, execution::unseq );
326 // Rethrow last exception caught
327 if ( eptr ) [[unlikely]]
328 {
329 rethrow_exception( eptr );
330 }
331 }
332}
