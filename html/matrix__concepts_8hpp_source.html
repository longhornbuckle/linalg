19// All matrix data types must satisfy tensor requirements
20// All matrix data types must be a rank 2 tensor
21// All matrix data types must support size / capacity const access functions
22template < class M >
23concept matrix_data = tensor_data<M> &&
24( M::extents_type::rank() == 2 ) &&
25requires( M& m ) // Functions
26{
27 { m.columns() } noexcept -> ::std::same_as< typename M::size_type >;
28 { m.rows() } noexcept -> ::std::same_as< typename M::size_type >;
29 { m.column_capacity() } noexcept -> ::std::same_as< typename M::size_type >;
30 { m.row_capacity() } noexcept -> ::std::same_as< typename M::size_type >;
31};

33// Matrix must be a matrix data
34// Matrix must be a tensor
35// All matrix data types must support unary functions: trans and conj
36// All matrix data types must support binary functions: multiplication
37template < class M >
38concept matrix = matrix_data<M> &&
39tensor<M> &&
40requires ( M m ) // Unary operations
41{
42 trans( m );
43 conj( m );
44} &&
45requires ( M m ) // Binary operations
46{
47 m * trans( m );
48};

50// Readable matrix data must A) be a matrix data
51// Readable matrix data must B) be a readable tensor data
52// Readable matrix data must C) provide index access
53// Readable matrix data must D) provide a two dimension const span representation
54// Readable matrix data must D) provide an implementation defined view of the underlying memory.
55// Readable matrix data must E) provide a const row view
56// Readable matrix data must F) provide a const column view
57// Readable matrix data must G) provide a const submatrix view
58template < class M >
59concept readable_matrix_data = matrix_data<M> && // A)
60readable_tensor_data<M> && // B)
61requires( const M& m, typename M::index_type index1, typename M::index_type index2 ) // C)
62{
63 #if LINALG_USE_BRACKET_OPERATOR
64 { m[index1,index2] } noexcept -> ::std::same_as<typename M::value_type>;
65 #endif
66 #if LINALG_USE_PAREN_OPERATOR
67 { m(index1,index2) } noexcept -> ::std::same_as<typename M::value_type>;
68 #endif
69 { m.at(index1,index2) } -> ::std::same_as<typename M::value_type>;
70} &&
71( M::span_type::extents_type::rank() == 2 ) && // D)
72requires( const M& m, typename M::index_type index ) // E)
73{
74 typename M::const_row_type;
75 { m.row(index) } -> ::std::same_as<typename M::const_row_type>;
76} &&
77readable_vector_data<typename M::const_row_type> &&
78requires( const M& m, typename M::index_type index ) // F)
79{
80 typename M::const_column_type;
81 { m.column(index) } -> ::std::same_as<typename M::const_column_type>;
82} &&
83readable_vector_data<typename M::const_column_type> &&
84requires( const M& m,
85 typename M::tuple_type rows,
86 typename M::tuple_type cols ) // G)
87{
88 { m.submatrix(rows,cols) } -> matrix;
89};
