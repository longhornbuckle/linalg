25// Vector must be a vector data

27// All vector types must support unary functions: trans and conj
28// All vector types must support binary functions: inner_prod and outer_prod

30concept vector = vector_data<V> &&
31tensor<V> &&
32requires ( V v ) // Unary operations

43// Readable vector data must A) be a vector data
44// Readable vector data must B) be a readable tensor data
45// Readable vector data must C) provide index access
46// Readable vector data must D) provide a one dimension const span representation
47// Readable vector data must E) provide a const subvector view
48template < class V >
49concept readable_vector_data = vector_data<V> && // A)
50readable_tensor_data<V> && // B)
51requires( const V& v, typename V::index_type index ) // C)

53 #if LINALG_USE_BRACKET_OPERATOR

55 #endif
56 #if LINALG_USE_PAREN_OPERATOR
57 { v(index) } noexcept -> ::std::same_as<typename V::value_type>;
58 #endif
59 { v.at(index) } -> ::std::same_as<typename V::value_type>;
60} &&
