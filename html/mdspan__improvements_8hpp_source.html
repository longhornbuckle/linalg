178 // Comparison operator
179 template <class OtherIndexType, size_t... OtherExtents>
180 MDSPAN_INLINE_FUNCTION friend constexpr bool
181 operator==(const extents &lhs,
182 const extents<OtherIndexType, OtherExtents...> &rhs) noexcept {
183 bool value = true;
184 for (size_type r = 0; r < m_rank; r++)
185 value &= rhs.extent(r) == lhs.extent(r);
186 return value;
187 }
