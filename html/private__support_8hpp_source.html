219template < class T >
220struct faux_index_iterator
221{
222 using value_type = T;
223 using difference_type = ::std::ptrdiff_t;
224 using reference = T&;
225 using pointer = T*;
226 using iterator_category = ::std::random_access_iterator_tag;
227 constexpr faux_index_iterator( T t ) noexcept : index(t) { }
228 constexpr faux_index_iterator& operator++ () noexcept { static_cast<void>(this->index++); return *this; }
229 constexpr faux_index_iterator& operator-- () noexcept { static_cast<void>(this->index--); return *this; }
230 constexpr faux_index_iterator& operator+= ( const difference_type n ) noexcept { static_cast<void>(this->index+=n); return *this; }
231 constexpr faux_index_iterator& operator-= ( const difference_type n ) noexcept { static_cast<void>(this->index-=n); return *this; }
232 [[nodiscard]] constexpr faux_index_iterator operator+ ( const difference_type n ) noexcept { return faux_index_iterator( this->index + n ); }
233 [[nodiscard]] constexpr faux_index_iterator operator- ( const difference_type n ) noexcept { return faux_index_iterator( this->index - n ); }
234 [[nodiscard]] constexpr difference_type operator - ( const faux_index_iterator& rhs ) noexcept { return difference_type( this->index - rhs.index ); }
235 [[nodiscard]] constexpr reference operator * () noexcept { return this->index; }
236 [[nodiscard]] constexpr pointer operator -> () noexcept { return &(this->index); }
237 [[nodiscard]] constexpr bool operator == ( const faux_index_iterator& rhs ) noexcept { return ( this->index == rhs.index ); }
238 [[nodiscard]] constexpr bool operator > ( const faux_index_iterator& rhs ) noexcept { return ( this->index > rhs.index ); }
239 [[nodiscard]] constexpr bool operator < ( const faux_index_iterator& rhs ) noexcept { return ( this->index < rhs.index ); }
240 [[nodiscard]] constexpr bool operator != ( const faux_index_iterator& rhs ) noexcept { return !( *this == rhs ); }
241 [[nodiscard]] constexpr bool operator >= ( const faux_index_iterator& rhs ) noexcept { return !( *this > rhs ); }
242 [[nodiscard]] constexpr bool operator <= ( const faux_index_iterator& rhs ) noexcept { return !( *this < rhs ); }
243 [[nodiscard]] constexpr value_type operator[] ( const difference_type n ) noexcept { return this->index + n; }
244 T index;
245};
