177// Faux Index Iterator allows indices to be used in std algorithms which take iterators
178//==================================================================================================
179template < class T >
180struct faux_index_iterator
181{
182 using value_type = T;
183 using difference_type = ptrdiff_t;
184 using reference = T&;
185 using pointer = T*;
186 using iterator_category = random_access_iterator_tag;
187 constexpr faux_index_iterator& operator++ () noexcept { static_cast<void>(this->index++); return *this; }
188 constexpr faux_index_iterator& operator-- () noexcept { static_cast<void>(this->index--); return *this; }
189 constexpr faux_index_iterator& operator+= ( const difference_type n ) noexcept { static_cast<void>(this->index+=n); return *this; }
190 constexpr faux_index_iterator& operator-= ( const difference_type n ) noexcept { static_cast<void>(this->index-=n); return *this; }
191 [[nodiscard]] constexpr faux_index_iterator operator+ ( const difference_type n ) noexcept { return faux_index_iterator( this->index + n ); }
192 [[nodiscard]] constexpr faux_index_iterator operator- ( const difference_type n ) noexcept { return faux_index_iterator( this->index - n ); }
193 [[nodiscard]] constexpr difference_type operator - ( const faux_index_iterator& rhs ) noexcept { return difference_type( this->index - rhs.index ); }
194 [[nodiscard]] constexpr reference operator * () noexcept { return this->index; }
195 [[nodiscard]] constexpr pointer operator -> () noexcept { return &(this->index); }
196 [[nodiscard]] constexpr bool operator == ( const faux_index_iterator& rhs ) noexcept { return ( this->index == rhs.index ); }
197 [[nodiscard]] constexpr bool operator > ( const faux_index_iterator& rhs ) noexcept { return ( this->index > rhs.index ); }
198 [[nodiscard]] constexpr bool operator < ( const faux_index_iterator& rhs ) noexcept { return ( this->index < rhs.index ); }
199 [[nodiscard]] constexpr bool operator != ( const faux_index_iterator& rhs ) noexcept { return !( *this == rhs ); }
200 [[nodiscard]] constexpr bool operator >= ( const faux_index_iterator& rhs ) noexcept { return !( *this > rhs ); }
201 [[nodiscard]] constexpr bool operator <= ( const faux_index_iterator& rhs ) noexcept { return !( *this < rhs ); }
202 [[nodiscard]] constexpr value_type operator[] ( const difference_type n ) noexcept { return this->index + n; }
203 T index;
204};

311template < class View,
312 class Lambda,
313 class ExecutionPolicy,
314 class IndexType,
315 IndexType ... BeforeIndices,
316 IndexType ... AfterIndices,
317 class ... IndicesType >
318constexpr void apply_all_strided3( View&& view,
319 Lambda&& lambda,
320 ExecutionPolicy&& execution_policy,
321 [[maybe_unused]] integer_sequence<IndexType,BeforeIndices...>,
322 [[maybe_unused]] integer_sequence<IndexType,AfterIndices...>,
323 tuple<IndicesType...>& indices )
324 noexcept( noexcept( lambda( get< BeforeIndices >( indices ) ...,
325 declval< decay_t< decltype( get< sizeof...(BeforeIndices) >( indices ) ) > >(),
326 get< AfterIndices + sizeof...(BeforeIndices) >( indices ) ... ) ) )
327{
328 constexpr IndexType index = sizeof...(BeforeIndices);
329 // If lambda expression is noexcept, then just attempt to call using whatever execution policy
330 if constexpr ( noexcept( lambda( get< BeforeIndices >( indices ) ...,
331 declval< decay_t< decltype( get< sizeof...(BeforeIndices) >( indices ) ) > >(),
332 get< AfterIndices + sizeof...(BeforeIndices) >( indices ) ... ) ) )
333 {
334 for_each( execution_policy,
335 faux_index_iterator<decay_t<decltype( get<index>(indices) )> >(0),
336 faux_index_iterator<decay_t<decltype( get<index>(indices) )> >(view.extent(index)),
337 [&lambda,&indices] ( decay_t<decltype( get<index>(indices) )> curr_index ) constexpr noexcept
338 {
339 lambda( get<BeforeIndices>(indices) ..., curr_index, get<AfterIndices+index>(indices) ... );
340 } );
341 }
342 else
343 {
344 // Cache the last exception to be thrown
345 exception_ptr eptr;
346 // Attempt lambda expression on each element
347 for_each( execution_policy,
348 faux_index_iterator<decay_t<decltype( get<index>(indices) )> >(0),
349 faux_index_iterator<decay_t<decltype( get<index>(indices) )> >(view.extent(index)),
350 [&lambda,&indices,&eptr] ( decay_t<decltype( get<index>(indices) )> curr_index ) constexpr noexcept
351 {
352 try { lambda( get<BeforeIndices>(indices) ..., curr_index, get<AfterIndices+index>(indices) ... ); }
353 catch ( ... ) { eptr = current_exception(); }
354 } );
355 // If exceptions were thrown, rethrow the last
356 if ( eptr ) [[unlikely]]
357 {
358 rethrow_exception( eptr );
359 }

535template < class View,
536 class Lambda,
537 class ExecutionPolicy,
538 class ... SizeType,
539 class ExtentsType,
540 ExtentsType FirstExtents,
541 ExtentsType ... Extents >
542constexpr void apply_all_impl( View&& view,
543 Lambda&& lambda,
544 ExecutionPolicy&& execution_policy,
545 [[maybe_unused]] integer_sequence<ExtentsType,FirstExtents,Extents...>,
546 SizeType ... indices )
547 noexcept( apply_all_impl_is_noexcept<View,Lambda,ExecutionPolicy,SizeType...>( integer_sequence<ExtentsType,FirstExtents,Extents...>{} ) )
548{
549 if constexpr ( sizeof...(Extents) == 0 )
550 {
551 apply_all_impl2( forward<View>( view ),
552 forward<Lambda>( lambda ),
553 forward<ExecutionPolicy>( execution_policy ),
554 FirstExtents, indices ... );
555 }
556 else
557 {
558 // Determines if the following lambda expression is noexcept
559 constexpr bool is_noexcept =
560 noexcept( apply_all_impl( forward<View>( view ),
561 forward<Lambda>( lambda ),
562 forward<ExecutionPolicy>( execution_policy ),
563 integer_sequence<ExtentsType,Extents...>{}, indices ..., declval<typename decay_t<View>::size_type>() ) );
564 // Lambda expression iterates over the next index
565 auto for_each_lambda = [&view,&lambda,&execution_policy,&indices...] ( typename decay_t<View>::size_type index )
566 constexpr noexcept( is_noexcept )
567 { apply_all_impl( forward<View>( view ),
568 forward<Lambda>( lambda ),
569 forward<ExecutionPolicy>( execution_policy ),
570 integer_sequence<ExtentsType,Extents...>{}, indices ..., index ); };
571 // If lambda expression is no except then no need to try to catch exception pointers
572 if constexpr ( is_noexcept )
573 {
574 for_each( execution_policy,
575 faux_index_iterator<typename decay_t<View>::size_type>( 0 ),
576 faux_index_iterator<typename decay_t<View>::size_type>( view.extent(FirstExtents) ),
577 for_each_lambda );
578 }
579 else
580 {
581 // Cache the last exception to be thrown
582 exception_ptr eptr;
583 // Attempt lambda expression
584 for_each( execution_policy,
585 faux_index_iterator<typename decay_t<View>::size_type>( 0 ),
586 faux_index_iterator<typename decay_t<View>::size_type>( view.extent(FirstExtents) ),
587 [ &for_each_lambda, &eptr ] ( typename decay_t<View>::size_type index ) constexpr noexcept
588 { try { for_each_lambda(index); } catch ( ... ) { eptr = current_exception(); } } );
589 // If exceptions were thrown, rethrow the last

690//==================================================================================================
691// Sufficient Extents tests if the first extents encompasses the second
692//==================================================================================================
693template < class SizeType, class OtherSizeType, size_t ... Extents, size_t ... OtherExtents >
694[[nodiscard]] constexpr bool sufficient_extents( const experimental::extents<SizeType,Extents...>& extents,
695 const experimental::extents<OtherSizeType,OtherExtents...>& other_extents ) noexcept
696{
697 if constexpr ( sizeof...(Extents) == sizeof...(OtherExtents) )
698 {
699 // Iterate over each dimension
700 bool sufficient = true;
701 for( auto dim = 0; ( dim < experimental::extents<SizeType,Extents...>::rank() && sufficient ); ++dim )
702 {
703 // Set to false if size is not large enough
704 sufficient = ( extents.extent( dim ) >= other_extents.extent( dim ) );
705 }
706 return sufficient;
707 }
708 else
709 {
710 return false;
711 }
712}
