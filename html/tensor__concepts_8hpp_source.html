19// All tensor data types must specify a value and index type
20// All tensor data types must specify an extents type which must be of type ::std::extents<...>
21// All tensor data types must support size / capacity const access functions
22template < class T >
23concept tensor_data = requires // Types
24{
25 typename T::value_type;
26 typename T::index_type;
27 typename T::size_type;
28 typename T::extents_type;
29} &&
30( detail::is_extents_v<typename T::extents_type> ) &&
31requires( T& t ) // Functions
32{
33 { t.size() } noexcept;
34 { t.capacity() } noexcept;
35} &&
36is_same_v< decltype( ::std::declval<T>().size() ), typename T::extents_type > &&
37is_same_v< decltype( ::std::declval<T>().capacity() ), typename T::extents_type >;

39// All tensor types are tensor data types
40// All tensor types must support scalar multiply and divide operations
41// All tensor types must support unary functions: negation
42// All tensor types must support binary functions: addition and subtraction
43template < class T >
44concept tensor = tensor_data<T> &&
45requires ( T& t, typename T::value_type v ) // Scalar functions
46{
47 v * t; // Return type should also be a tensor or a derived type
48 t * v; // Return type should also be a tensor or a derived type
49 t / v; // Return type should also be a tensor or a derived type
50} &&
51requires ( T t ) // Unary operations
52{
53 -t; // Return type should also be a tensor or a derived type
54} &&
55requires( T t ) // Binary operations
56{
57 t + t; // Return type should also be a tensor or a derived type
58 t - t; // Return type should also be a tensor or a derived type
59};

61// Readable tensor data must A) be a tensor data
62// Readable tensor data must B) provide a tuple type of extents_type::rank() index_type
63// Readable tensor data must C) provide index access
64// Readable tensor data must D) provide a const span representation
65// Readable tensor data must E) provide an implementation defined view of the underlying memory.
66// Readable tensor data must F) provide a const subtensor view

167// Fixed size tensor data
168// A fixed size tensor data must be compile-time constructible
169// A fixed size tensor data must support constant expression size functions.
170// A fixed size tensor data must support constant expression capacity functions.
171// A fixed size tensor data must have equivalent size and capacity.
172template < class T >
173concept fixed_size_tensor_data = tensor_data<T> &&
174detail::extents_is_static_v<typename T::extents_type> &&
175detail::is_constexpr( []{ T(); } ) &&
176detail::is_constexpr( []{ decltype( ::std::declval<T>().size() ) nodiscard_warning = T().size(); } ) &&
177detail::is_constexpr( []{ decltype( ::std::declval<T>().capacity() ) nodiscard_warning = T().capacity(); } ) &&
178( T().size() == T().capacity() );

189// Tensor may be constructible
190// Enforces both types are tensor which have constructible elements
191// and both have potentially compatible extents
192template < class From, class To >
193concept tensor_may_be_constructible =
194tensor<From> && tensor<To> &&
195( ::std::is_constructible_v< typename From::value_type, typename To::value_type > ) &&
196detail::extents_may_be_equal_v< typename From::extents_type, typename To::extents_type >;

198// View may be constructible to tensor
199// Enforces view is an mdspan of rank equal to tensor with elements
200// Each extent must be the same or dynamic
201// View element must be constructible to the tensor elements
202template < class From, class To >
203concept view_may_be_constructible_to_tensor =
204detail::is_mdspan_v<From> && tensor<To> &&
205( ::std::is_constructible_v<typename From::value_type,typename To::value_type> ) &&
206detail::extents_may_be_equal_v< typename From::extents_type, typename To::extents_type >;

208// View is constructible to tensor
209// Enforces view is an mdspan of rank equal to tensor with elements
210// Each extent must be the same (and not dynamic)
211// View element must be constructible to the tensor elements
212template < class From, class To >
213concept view_is_constructible_to_tensor =
214detail::is_mdspan_v<From> && tensor<To> &&
215( ::std::is_constructible_v<typename From::value_type,typename To::value_type> ) &&
216detail::extents_are_equal_v< typename From::extents_type, typename To::extents_type >;

218// View is nothrow constructible to tensor
219// Enforces view is an mdspan of rank equal to tensor with elements
220// Each extent must be the same (and not dynamic)
221// View element must be nothrow constructible to the tensor elements
222template < class From, class To >
